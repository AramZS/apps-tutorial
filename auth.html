Authenticate a user with Facebook

Here, we'll attempt to authenticate a user from Facebook.

The Javascript

<script src="js/jquery.min.js"></script>
<script>
    $(document).ready(function() {
        /* Instantiate the global sp object; include models & views */
        sp = getSpotifyApi(1);
        var auth = sp.require("sp://import/scripts/api/auth");

         auth.authenticateWithFacebook('YOUR_APP_ID', ['user_about_me'], {
            onSuccess : function(accessToken, ttl) {
                $('#status > p').append("<p>" + accessToken + "</p>");
            }, onFailure : function(error) {
                console.log("Failed with error: " + error);
            }, onComplete : function() { 
                // More JS!
            }
        });                
    });
</script>

The HTML

<body>
    <div id="status"><p>If you've successfully authenticated with
    Facebook, this should be your access token:</p></div>
</body>
